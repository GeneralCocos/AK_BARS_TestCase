3)	SELECT CardType, (100.0 * SUM(CASE WHEN Exitet = 1 THEN 1 ELSE 0 END)/ 
COUNT(*)) AS exited_percent
FROM Clients
GROUP BY CardType
ORDER BY exited_percent

4)	SELECT CustomerId, Geography, EstimatedSalary,
(SELECT AVG(EstimatedSalary)
FROM Clients
GROUP BY Geography) AS avg_salary
	FROM Clients 

5)	WITH ranked AS(
SELECT Geography, Gender EstimatedSalary,
ROW_NUMBER() OVER (PARTITION BY Geography ORDER BY EstimatedSalary DESC) AS rn
FROM Clients)

SELECT Geography,
	SUM(CASE WHEN Gender = ‘Female’ THEN 1 ELSE 0 END) AS top10_female
	SUM(CASE WHEN Gender = ‘Male’ THEN 1 ELSE 0 END) AS top10_male
FROM ranked 
WHERE rn <= 10
GROUP BY Geography
HAVING SUM(CASE WHEN Gender = ‘Female’ THEN 1 ELSE 0 END) AS top10_female >
	SUM(CASE WHEN Gender = ‘Male’ THEN 1 ELSE 0 END) AS top10_male
ORDER BY Geography

FROM Clients
WHERE (SELECT COUNT(DISTINCT Сard Type)
	FROM Clients) < 4
